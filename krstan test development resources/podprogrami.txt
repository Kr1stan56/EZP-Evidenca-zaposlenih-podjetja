-- Primer PL/pgSQL funkcije v PostgreSQL
CREATE OR REPLACE FUNCTION login_admin(p_username VARCHAR)
RETURNS TABLE(password VARCHAR) AS $$
BEGIN
    RETURN QUERY
    SELECT u.password
    FROM admin_users u
    WHERE u.username = p_username;
END;
$$ LANGUAGE plpgsql;

CREATE OR REPLACE FUNCTION register_user(
    p_username TEXT,
    p_password_hash TEXT,
    p_email TEXT
)
RETURNS BOOLEAN AS $$
BEGIN
    IF EXISTS (SELECT 1 FROM admin_users WHERE username = p_username) THEN
        RETURN FALSE;
    END IF;

    INSERT INTO admin_users(username, password, email)
    VALUES (p_username, p_password_hash, p_email);

    RETURN TRUE;
END;
$$ LANGUAGE plpgsql;



CREATE OR REPLACE FUNCTION get_all_employees()
RETURNS TABLE(
    id INT,
    ime VARCHAR,
    priimek VARCHAR,
    email VARCHAR,
    telefon VARCHAR,
    placa REAL,
    datum_zasposlitve DATE,
    delovno_mesto VARCHAR,
    oddelek VARCHAR
) AS $$
BEGIN
    RETURN QUERY
    SELECT
        z.id,
        z.ime,
        z.priimek,
        z.email,
        z.telefon,
        z.placa,
        z.datum_zasposlitve,
        dm.naziv as delovno_mesto,
        o.ime as oddelek
    FROM zaposleni z
    JOIN delovna_mesta dm ON z.delovno_mesto_id = dm.id
    JOIN oddelki o ON z.oddelek_id = o.id
    ORDER BY z.priimek, z.ime;
END;
$$ LANGUAGE plpgsql;

CREATE OR REPLACE FUNCTION add_employee(
    p_ime VARCHAR,
    p_priimek VARCHAR,
    p_email VARCHAR,
    p_telefon VARCHAR,
    p_placa REAL,
    p_datum DATE,
    p_kraj_id INT,
    p_oddelek_id INT
) RETURNS VOID AS $$
BEGIN
    INSERT INTO zaposleni (ime, priimek, email, telefon, placa, datum_zasposlitve, kraj_id, oddelek_id, delovno_mesto_id)
    VALUES (p_ime, p_priimek, p_email, p_telefon, p_placa, p_datum, p_kraj_id, p_oddelek_id, 1);
END;
$$ LANGUAGE plpgsql;

CREATE OR REPLACE FUNCTION delete_employee(p_id INT)
RETURNS VOID AS $$
BEGIN
    DELETE FROM zaposleni WHERE id = p_id;
END;
$$ LANGUAGE plpgsql;
